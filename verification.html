  <script>

    function getQueryParams() {

      const params = new URLSearchParams(window.location.search);

      return {

        applicant: params.get('applicant'),

        docName: params.get('docName'),

        refNumber: params.get('refNumber'),

        status: params.get('status'),

        validFrom: params.get('validFrom'),

        validTo: params.get('validTo'),

        cropName: params.get('cropName')

      };

    }


    function isValidDate(dateStr) {

      return !isNaN(new Date(dateStr).getTime());

    }


    const data = getQueryParams();


    // Check required fields

    if (

      data.applicant && data.docName && data.refNumber &&

      data.status && data.validFrom && data.validTo &&

      isValidDate(data.validFrom) && isValidDate(data.validTo)

    ) {

      // Format dates

      const options = { year: 'numeric', month: 'long', day: 'numeric' };

      const validFromStr = new Date(data.validFrom).toLocaleDateString('en-GB', options);

      const validToStr = new Date(data.validTo).toLocaleDateString('en-GB', options);


      const permitDisplay = document.getElementById('permitDisplay');

      permitDisplay.innerHTML = `

        <p><span class="highlight">Applicant Name:</span> ${data.applicant}</p>

        <p><span class="highlight">Document Name:</span> ${data.docName}</p>

        <p><span class="highlight">Reference Number:</span> ${data.refNumber}</p>

        <p><span class="highlight">Current Status:</span> ${data.status}</p>

        <p><span class="highlight">Valid From:</span> ${validFromStr}</p>

        <p><span class="highlight">Valid To:</span> ${validToStr}</p>

        <p><span class="highlight">Crop/Product Name:</span> ${data.cropName || '(N/A)'}</p>

      `;

      permitDisplay.style.display = 'inline-block';

    } else {

      document.getElementById('error').style.display = 'block';

    }

  </script>
